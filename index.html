function doPost(e) {
  var sheet = SpreadsheetApp
    .getActiveSpreadsheet()
    .getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  var row = [
    data.timestamp, data.branch, data.sales,
    data.org, data.manager, data.classes,
    data.devices, data.isp, data.plan,
    data.routerModel, data.routerRAM,
    data.routerRamOk, data.loc, data.year,
    data.dist, data.disc, data.speed,
    data.grade, data.memo
  ];
  sheet.appendRow(row);
  return ContentService
    .createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService
